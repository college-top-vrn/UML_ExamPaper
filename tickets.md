# Аттестация по теме «Объектно-ориентированное проектирование, язык UML и паттерны проектирования»

## Общая информация

Аттестация состоит из 20 билетов. Каждый билет включает три части:

1. **Тест** — 5 вопросов с вводом ответа текстом (открытые вопросы).
2. **Практическое задание 1** — создание UML-диаграммы по текстовому описанию.
3. **Практическое задание 2** — написание программного кода на языке **C#** или **Kotlin** (по выбору студента) с реализацией паттерна проектирования.

---

## Критерии оценивания

### 1. Оценивание тестовой части

- Всего **5 вопросов** открытого типа (ввод текста).
- Каждый правильный ответ оценивается в **1 балл**.
- Ответ считается правильным, если он полностью совпадает с эталоном (регистр не учитывается, лишние пробелы игнорируются).
- Неправильный ответ или отсутствие ответа — **0 баллов**.
- **Оценка за тест** — сумма набранных баллов (от **0 до 5**).

### 2. Оценивание практических заданий

Каждое практическое задание (UML-диаграмма и написание кода) оценивается экспертно по **пятибалльной шкале**:

| Отметка | Критерий                                                                                             |
|---------|-------------------------------------------------------------------------------------------------------|
| **5**   | Задание выполнено полностью и верно, все требования соблюдены, ошибки отсутствуют.                     |
| **4**   | Имеются незначительные неточности, не влияющие на общее понимание и корректность решения.              |
| **3**   | Присутствуют существенные ошибки, но основная идея и структура задания реализованы.                     |
| **2**   | Задание не выполнено либо выполнено с грубыми принципиальными ошибками, демонстрирующими непонимание темы. |

Обозначения:
- \(M_T\) — оценка за тест (0–5)
- \(M_{P1}\) — оценка за практическое задание 1 (UML) (2–5)
- \(M_{P2}\) — оценка за практическое задание 2 (код) (2–5)

### 3. Вычисление итоговой оценки за билет (\(M_{\text{итог}}\))

Итоговая оценка вычисляется как **среднее арифметическое** трёх оценок:

```math
M_{\text{итог}} = \frac{M_T + M_{P1} + M_{P2}}{3}
```

**Правила округления:**  
Округление производится по математическим правилам (0,5 и выше — в бо́льшую сторону).

**Пример расчёта:**
- Тест: 4 балла
- UML: 5
- Код: 4

```math
\text{Среднее} = \frac{4 + 5 + 4}{3} = \frac{13}{3} \approx 4,33 \quad \rightarrow \quad \text{Итоговая оценка: 4 (хорошо)}
```

**Примечание:** если какое-либо из практических заданий не выполнено (оценка «2»), это автоматически снижает средний балл, что отражается на итоговой оценке. Отдельного правила о блокировке положительной оценки нет — всё определяется средним арифметическим.

```csharp
using System;
using System.Collections.Generic;

// Интерфейс наблюдателя
public interface IObserver
{
    void Update(string news);
}

// Интерфейс наблюдаемого объекта
public interface IObservable
{
    void Subscribe(IObserver observer);
    void Unsubscribe(IObserver observer);
    void NotifyObservers();
}

// Издатель новостей
public class NewsAgency : IObservable
{
    private List<IObserver> _observers = new List<IObserver>();
    private string _latestNews;

    public void SetNews(string news)
    {
        _latestNews = news;
        NotifyObservers();
    }

    public void Subscribe(IObserver observer)
    {
        _observers.Add(observer);
    }

    public void Unsubscribe(IObserver observer)
    {
        _observers.Remove(observer);
    }

    public void NotifyObservers()
    {
        foreach (var observer in _observers)
            observer.Update(_latestNews);
    }
}

// Подписчик
public class Subscriber : IObserver
{
    private string _name;

    public Subscriber(string name)
    {
        _name = name;
    }

    public void Update(string news)
    {
        Console.WriteLine($"{_name} получил новость: {news}");
    }
}

// Демонстрация
class Program
{
    static void Main()
    {
        NewsAgency agency = new NewsAgency();
        Subscriber sub1 = new Subscriber("Иван");
        Subscriber sub2 = new Subscriber("Мария");

        agency.Subscribe(sub1);
        agency.Subscribe(sub2);

        agency.SetNews("Первая новость!");

        agency.Unsubscribe(sub1);
        agency.SetNews("Вторая новость (только для Марии).");

        Console.ReadKey();
    }
}
```

```csharp
public interface INotifier
{
    void Send(string message);

    }

public abstract class NotifierDecorator : INotifier

    protected INotifier _wrapped;

    public NotifierDecorator(INotifier wrapped)
    {

    }

    public virtual void Send(string message)
    {
        _wrapped.Send(message);
    }

    }

    public override void Send(string message)
    {
        _wrapped.Send($"[Email] {message}");
    }
}

class Program
{
    static void Main()
    {
        INotifier notifier = new BaseNotifier();
        notifier = new SMSDecorator(notifier);
```

```csharp
using System;
using System.Collections.Generic;

// Стратегия уведомления
public interface INotificationStrategy
{
    void Notify(string message);
}

public class ConsoleNotification : INotificationStrategy
{
    public void Notify(string message) => Console.WriteLine($"[Console] {message}");
}

public class EmailNotification : INotificationStrategy
{
    private string _email;
    public EmailNotification(string email) => _email = email;
    public void Notify(string message) => Console.WriteLine($"[Email to {_email}] {message}");
}

// Наблюдатель
public interface IObserver
{
    void Update(string stockSymbol, decimal price);
}

public class Investor : IObserver
{
    private string _name;
    private INotificationStrategy _strategy;

    public Investor(string name, INotificationStrategy strategy)
    {
        _name = name;
        _strategy = strategy;
    }

    public void Update(string stockSymbol, decimal price)
    {
        string message = $"{_name} уведомлён: {stockSymbol} теперь {price}";
        _strategy.Notify(message);
    }
}

// Наблюдаемый объект
public class Stock
{
    private string _symbol;
    private decimal _price;
    private List<IObserver> _observers = new List<IObserver>();

    public Stock(string symbol, decimal initialPrice)
    {
        _symbol = symbol;
        _price = initialPrice;
    }

    public void Attach(IObserver observer) => _observers.Add(observer);
    public void Detach(IObserver observer) => _observers.Remove(observer);

    public decimal Price
    {
        get => _price;
        set
        {
            if (_price != value)
            {
                _price = value;
                Notify();
            }
        }
    }

    private void Notify()
    {
        foreach (var obs in _observers)
            obs.Update(_symbol, _price);
    }
}

class Program
{
    static void Main()
    {
        Stock apple = new Stock("AAPL", 150m);

        Investor investor1 = new Investor("Иван", new ConsoleNotification());
        Investor investor2 = new Investor("Мария", new EmailNotification("maria@example.com"));

        apple.Attach(investor1);
        apple.Attach(investor2);

        apple.Price = 155m;
        apple.Price = 160m;

        apple.Detach(investor1);
        apple.Price = 162m;

        Console.ReadKey();
    }
}
```
